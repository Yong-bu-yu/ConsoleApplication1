//创建一个新的Customer对象数组
private ArrayList CustomerArray=new ArrayList();
private void FillMyTreeView()
{
  //添加1000个customers到Customer对象数组
  for(int x=0;x<1000;x++)
  {
    customerArray.Add(new Customer("Customer"+x.Tostring)));
  }
  //为每一个Customer对象添加订单
  foreach(Customer customer1 in customerArray)
  {
    for(int y=0;y<15;y++)
    {
      customer1.CustomerOrders.Add(new Order("Order"+y.Tostring)));
    }
  }
  //当创建TreeNodes显示等待的鼠标
  Cursor.Current=new Cursor("C:\\Cursors\\MyWait.cur");
  treeView1.BeginUpdate();
  treeView1.Nodes.Clear();
  //在数组中为每一个Customer对象添加一个树根结点
  foreach(Customer customer2 in customerArray)
  {
    treeView1.Nodes.Add(new TreeNode(customer2.CustomerName));
    //为当前Customer对象添加一个孩子结点
    foreach(Order order1 in customer2.CustomerOrders)
    {
      treeView1.Nodes[customerArray.IndexOf(customer2)].Nodes.Add(new TreeNode(customer2.CustomerName+"."+order1.OrderID));
    }
  }
  Cursor.Current=Cursors.Default;
  //开始重画TreeView
  treeView1.EndUpdate();
}
